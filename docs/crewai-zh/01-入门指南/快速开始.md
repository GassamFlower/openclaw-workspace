# CrewAI 快速开始

> 5 分钟内跑通你的第一个 Crew

## 🚀 第一步: Hello Crew

### 最简单的 Crew

```python
from crewai import Agent, Task, Crew

# 1. 创建 Agent
researcher = Agent(
    role="研究员",
    goal="研究技术",
    backstory="你是一位科技专家",
    verbose=True
)

# 2. 创建 Task
task = Task(
    description="用一句话介绍你自己",
    agent=researcher
)

# 3. 创建 Crew 并运行
crew = Crew(
    agents=[researcher],
    tasks=[task],
    verbose=True
)

result = crew.kickoff()
print(result)
```

**预期输出**:
```
🚀 Task Starting...
Task completed successfully.
Your task: 用一句话介绍你自己

你好!我是一个基于 AI 的智能助手,可以帮您回答问题、完成任务。
```

## 🎯 第二步: 创建你的第一个 Crew

### 多 Agent 团队

```python
from crewai import Agent, Task, Crew

# 研究员 Agent
researcher = Agent(
    role="研究员",
    goal="收集最新科技资讯",
    backstory="你是一位资深的科技研究员",
    verbose=True
)

# 写作 Agent
writer = Agent(
    role="写作专家",
    goal="撰写高质量文章",
    backstory="你是一位经验丰富的科技作家",
    verbose=True
)

# 编辑 Agent
editor = Agent(
    role="编辑",
    goal="确保文章质量",
    backstory="你是一位严格的编辑",
    verbose=True
)

# 任务 1: 研究
research_task = Task(
    description="搜索最新的 AI 技术进展",
    expected_output="一份包含 3 个最新 AI 技术的列表",
    agent=researcher
)

# 任务 2: 写作
writing_task = Task(
    description="根据研究结果撰写一篇 500 字的文章",
    expected_output="一篇结构清晰的技术文章",
    agent=writer
)

# 任务 3: 编辑
editing_task = Task(
    description="检查并优化文章",
    expected_output="润色后的文章",
    agent=editor
)

# 创建并运行 Crew
crew = Crew(
    agents=[researcher, writer, editor],
    tasks=[research_task, writing_task, editing_task],
    verbose=True
)

result = crew.kickoff()
print(result)
```

## 🔥 第三步: 使用工具

### 给 Agent 赋予能力

```python
from crewai import Agent, Task, Crew
from crewai_tools import SerperDevTool

# 创建搜索工具
search_tool = SerperDevTool()

# 创建 Agent
researcher = Agent(
    role="研究员",
    goal="研究信息",
    tools=[search_tool],
    verbose=True
)

# 创建任务
task = Task(
    description="搜索 'CrewAI' 的最新信息",
    expected_output="CrewAI 的最新介绍",
    agent=researcher
)

# 运行
crew = Crew(agents=[researcher], tasks=[task], verbose=True)
result = crew.kickoff()
```

## 📊 第四步: 不同流程

### Sequential (顺序执行)

```python
# 任务按顺序执行
crew = Crew(
    agents=[agent1, agent2, agent3],
    tasks=[task1, task2, task3],
    process="sequential",  # 默认
    verbose=True
)
```

### Hierarchical (层级管理)

```python
# 添加 Manager Agent
manager = Agent(
    role="项目经理",
    goal="协调团队完成任务",
    backstory="你擅长管理",
    allow_delegation=True,  # 允许委派
    verbose=True
)

crew = Crew(
    agents=[manager, agent1, agent2],
    tasks=[task1, task2, task3],
    process="hierarchical",
    verbose=True
)
```

## 💡 第五步: 多轮交互

### 使用 Memory

```python
from crewai import Agent, Task, Crew
from crewai.memory import BufferMemory

# 创建记忆
memory = BufferMemory()

# 创建 Agent
agent = Agent(
    role="助手",
    goal="记住用户的偏好",
    backstory="你是一个贴心的助手",
    verbose=True,
    memory=True
)

# 任务
task = Task(
    description="记住用户的喜好",
    agent=agent
)

# 运行 Crew
crew = Crew(
    agents=[agent],
    tasks=[task],
    memory=memory,
    verbose=True
)

# 第 1 轮
result1 = crew.kickoff({"user_input": "我喜欢 Python"})

# 第 2 轮 - Agent 记得
result2 = crew.kickoff({"user_input": "我还有什么偏好?"})
```

## 🎮 实战示例: 客服系统

```python
from crewai import Agent, Task, Crew
from crewai_tools import SerperDevTool

# 搜索工具
search_tool = SerperDevTool()

# 分类 Agent
classifier = Agent(
    role="问题分类员",
    goal="判断问题类型",
    backstory="你擅长分类问题",
    tools=[search_tool],
    verbose=True
)

# 解决 Agent
solver = Agent(
    role="问题解决者",
    goal="解决问题",
    backstory="你善于解决常见问题",
    tools=[search_tool],
    verbose=True
)

# 分类任务
classification_task = Task(
    description="判断用户问题类型",
    expected_output="问题类型",
    agent=classifier
)

# 解决任务
solving_task = Task(
    description="解决用户问题",
    expected_output="解决方案",
    agent=solver
)

# 创建 Crew
crew = Crew(
    agents=[classifier, solver],
    tasks=[classification_task, solving_task],
    process="sequential",
    verbose=True
)

# 测试
result = crew.kickoff({"user_input": "如何安装 Python?"})
```

## 📝 完整示例: 内容创作团队

```python
from crewai import Agent, Task, Crew
from crewai_tools import SerperDevTool

# 工具
search_tool = SerperDevTool()

# Agents
researcher = Agent(
    role="研究员",
    goal="收集资料",
    backstory="你是一位专业研究员",
    tools=[search_tool],
    verbose=True
)

writer = Agent(
    role="写作者",
    goal="撰写内容",
    backstory="你是一位优秀的作家",
    verbose=True
)

editor = Agent(
    role="编辑",
    goal="审核内容",
    backstory="你是一位严格的编辑",
    verbose=True
)

# Tasks
research = Task(
    description="搜索 AI 在医疗领域的应用",
    expected_output="3 个关键应用点",
    agent=researcher
)

write = Task(
    description="撰写一篇 800 字文章",
    expected_output="完整文章",
    agent=writer
)

edit = Task(
    description="检查并优化文章",
    expected_output="最终文章",
    agent=editor
)

# Crew
crew = Crew(
    agents=[researcher, writer, editor],
    tasks=[research, write, edit],
    process="sequential",
    verbose=True
)

# 运行
result = crew.kickoff()
print(result)
```

## 💡 学习要点

### 1. Agent 设计

```python
# ✅ 好的 Agent 定义
agent = Agent(
    role="明确的角色",
    goal="清晰的目标",
    backstory="专业的背景",
    tools=[...],  # 必要的工具
    verbose=True  # 便于调试
)

# ❌ 糟糕的 Agent 定义
agent = Agent(tools=[...])  # 缺少角色和目标
```

### 2. Task 设计

```python
# ✅ 好的 Task 定义
task = Task(
    description="清晰的任务描述",
    expected_output="具体的期望输出",
    agent=agent,
    tools=[...]  # 必要的工具
)

# ❌ 糟糕的 Task 定义
task = Task(
    description="",  # 描述不清
    agent=agent
)
```
