## 💻 基础调试技巧

### 1. 启用 Verbose 模式

```python
from langchain_openai import ChatOpenAI
from langchain.chains import ConversationChain

# 启用详细模式
llm = ChatOpenAI(model="gpt-4", verbose=True)

# 创建 Chain
conversation = ConversationChain(llm=llm, verbose=True)

# 运行
response = conversation.predict(input="你好")
````

### 2. 打印输入输出

```python
from langchain.chains import LLMChain

# 创建 Chain
chain = LLMChain(llm=llm, prompt=prompt)

# 运行并打印输入输出
input_data = {"language": "中文", "topic": "AI"}
result = chain.run(**input_data)

print("输入:", input_data)
print("输出:", result)
````

### 3. 记录执行过程

```python
import logging

# 配置日志
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

def run_with_logging(chain, input_data):
    logger.info(f"开始执行: {input_data}")

    try:
        result = chain.run(**input_data)
        logger.info(f"执行成功: {result}")
        return result
    except Exception as e:
        logger.error(f"执行失败: {e}")
        raise
`````

## 🎯 常见问题调试

### 问题 1: Agent 不知道用什么工具

```python
# ❌ 问题: Agent 不知道调用哪个工具
# 解决方案: 明确的描述

@tool
def search_tool(query: str) -> str:
    """搜索信息。
    用于回答关于{topic}的问题。
    例如: '搜索人工智能的最新进展'
    """
    return f"搜索: {query}"

# ✅ 正确的工具描述
tools = [
    Tool(
        name="Search",
        func=search_tool,
        description="搜索信息,用于回答关于技术、产品、服务等问题"
    )
]
````

### 问题 2: Agent 返回空答案

```python
# ❌ 问题: Agent 不回答问题
# 解决方案: 检查 Chain 配置

# 检查 Prompt
print(prompt.format_messages())

# 检查 Chain
print(chain.run(input="test"))

# 检查 LLM
response = llm.invoke("test")
print(response.content)
````

### 问题 3: 性能问题

```python
# ❌ 问题: 响应很慢
# 解决方案: 优化和缓存

# 1. 使用缓存
from langchain.cache import InMemoryCache
from langchain.globals import set_llm_cache
set_llm_cache(InMemoryCache())

# 2. 减少重复调用
# ❌ 每次都调用 LLM
result1 = llm.invoke("你好")
result2 = llm.invoke("你好")

# ✅ 使用缓存
result1 = llm.invoke("你好")
result2 = cached_llm.invoke("你好")  # 从缓存读取
````

### 问题 4: 内存问题

```python
# ❌ 问题: 长对话导致内存溢出
# 解决方案: 使用窗口记忆

from langchain.memory import ConversationBufferWindowMemory

memory = ConversationBufferWindowMemory(
    k=4,  # 只保留最近 4 轮
    return_messages=True
)
`````

### 2. 响应时间分析

```python
import time

def time_it(func):
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"耗时: {end - start:.2f}秒")
        return result
    return wrapper

@time_it
def my_function():
    return llm.invoke("问题")

my_function()
````

### 3. 并行执行分析

```python
from langchain_core.runnables import RunnableParallel

# 并行执行
start = time.time()
result = RunnableParallel(
    chain1=chain1,
    chain2=chain2,
    chain3=chain3
).run(input="data")
end = time.time()

print(f"并行执行耗时: {end - start:.2f}秒")
````

## 🚨 错误处理

### 1. 基本错误处理

```python
try:
    result = chain.run(input="data")
except Exception as e:
    print(f"错误: {e}")
    # 回退方案
    result = fallback_chain.run(input="data")
````

### 2. 重试机制

```python
from tenacity import retry, stop_after_attempt, wait_exponential

@retry(
    stop=stop_after_attempt(3),
    wait=wait_exponential(multiplier=1, min=4, max=60)
)
def call_llm_with_retry(prompt):
    return llm.invoke(prompt)
````
