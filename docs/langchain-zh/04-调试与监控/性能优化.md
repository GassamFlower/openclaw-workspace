## 💻 优化技巧

### 1. 使用缓存

```python
from langchain.cache import InMemoryCache
from langchain.globals import set_llm_cache

# 启用缓存
set_llm_cache(InMemoryCache())

# 重复调用会从缓存读取
result1 = llm.invoke("问题 1")
result2 = llm.invoke("问题 1")  # 从缓存读取
````

### 2. 批量处理

```python
# ❌ 逐个处理
for item in items:
    result = llm.invoke(item)
    results.append(result)

# ✅ 批量处理
results = llm.batch([item for item in items])
````

### 3. 并行执行

```python
from langchain_core.runnables import RunnableParallel

# 并行执行多个 Chain
results = RunnableParallel(
    chain1=chain1,
    chain2=chain2,
    chain3=chain3
).run(input="data")

# 所有 Chain 同时执行
````

import time

def measure_time(func):
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"耗时: {end - start:.2f}秒")
        return result
    return wrapper

@measure_time
def my_function():
    return llm.invoke("问题")

my_function()
````

## 🚀 企业级优化

### 1. 模型缓存

```python
# 使用 Redis 缓存
from redis import Redis

redis_client = Redis(host='localhost', port=6379)

def get_cached_response(prompt):
    cache_key = f"response:{prompt}"
    cached = redis_client.get(cache_key)

    if cached:
        return cached.decode()

    response = llm.invoke(prompt)
    redis_client.setex(cache_key, 3600, response.content)
    return response.content
````

### 2. 负载均衡

```python
# 使用多个 LLM 实例
from langchain_openai import ChatOpenAI

class LLMPool:
    def __init__(self, models):
        self.models = models
        self.index = 0

    def get_model(self):
        model = self.models[self.index]
        self.index = (self.index + 1) % len(self.models)
        return model

    def invoke(self, prompt):
        model = self.get_model()
        return model.invoke(prompt)

llm_pool = LLMPool([
    ChatOpenAI(model="gpt-4"),
    ChatOpenAI(model="gpt-3.5-turbo"),
])
````

### 3. 异步执行

```python
import asyncio

async def process_async(prompt):
    # 异步调用
    result = await llm.ainvoke(prompt)
    return result

async def batch_process_async(prompts):
    # 批量异步处理
    tasks = [process_async(prompt) for prompt in prompts]
    return await asyncio.gather(*tasks)

# 运行
results = asyncio.run(batch_process_async(prompts))
`````
