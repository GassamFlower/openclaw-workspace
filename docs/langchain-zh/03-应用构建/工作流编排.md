# 工作流编排 - LangChain 高级应用

> 复杂 AI 工作流的最佳实践

## 🎯 什么是工作流编排?

### 简单理解

**工作流编排 = 把多个步骤有序地组合起来**

LangChain 提供多种 Chain 类型,你可以像搭积木一样组合各种组件:

```
用户输入 → Step 1 → Step 2 → Step 3 → 结果
```

**类比**:
- 就像制作一道菜
- Step 1: 切菜
- Step 2: 烹饪
- Step 3: 装盘
- 最后: 上菜

## 💻 Chain 类型

### 1. LLM Chain (最基础)

```python
from langchain_openai import ChatOpenAI
from langchain.prompts import ChatPromptTemplate

# 创建 LLM
llm = ChatOpenAI(model="gpt-4", temperature=0)

# 创建提示词模板
prompt = ChatPromptTemplate.from_template(
    "用 {language} 解释 {topic}"
)

# 创建 Chain
chain = LLMChain(llm=llm, prompt=prompt)

# 使用
result = chain.run(language="中文", topic="LangChain")
````

### 2. Sequential Chain (顺序链)

```python
from langchain.chains import SequentialChain

# Chain 1: 搜索信息
search_chain = LLMChain(llm=llm, prompt=search_prompt, output_key="search_result")

# Chain 2: 总结信息
summary_chain = LLMChain(llm=llm, prompt=summary_prompt, output_key="summary")

# Chain 3: 写作
writing_chain = LLMChain(llm=llm, prompt=writing_prompt, output_key="article")

# 创建顺序链
seq_chain = SequentialChain(
    chains=[search_chain, summary_chain, writing_chain],
    input_variables=["topic"],
    output_variables=["search_result", "summary", "article"]
)

# 使用
result = seq_chain(topic="人工智能")
````

### 3. Router Chain (路由链)

```python
from langchain.chains import RouterChain

# 定义不同类型的 Chain
research_chain = LLMChain(llm=llm, prompt=research_prompt)
code_chain = LLMChain(llm=llm, prompt=code_prompt)
chat_chain = LLMChain(llm=llm, prompt=chat_prompt)

# 创建路由 Chain
router_chain = RouterChain(
    llm=llm,
    router_prompt="根据问题类型选择合适的 Chain",
    chains={
        "research": research_chain,
        "code": code_chain,
        "chat": chat_chain
    }
)

# 使用
result = router_chain.run("帮我写一段代码")
````

### 4. Transform Chain (转换链)

```python
from langchain.chains import TransformChain

# 自定义转换函数
def transform_func(inputs):
    text = inputs["text"]
    # 转换逻辑
    return {"transformed_text": text.upper()}

# 创建转换链
transform_chain = TransformChain(
    output_variables=["transformed_text"],
    transform=transform_func
)

# 使用
result = transform_chain.run(text="hello world")
# 结果: "HELLO WORLD"
````

### 5. Composite Chain (组合链)

```python
from langchain.chains import CompositeChain

# 多个输入变量
chain1 = LLMChain(llm=llm, prompt=prompt1)
chain2 = LLMChain(llm=llm, prompt=prompt2)

# 组合链
composite_chain = CompositeChain(
    chains=[chain1, chain2],
    input_variables=["input1", "input2"],
    output_variables=["output1", "output2"]
)

# 使用
result = composite_chain.run(input1="数据", input2="分析")
`````

## 🎮 实战示例

### 示例 1: 内容生成工作流

```python
from langchain.chains import SequentialChain

# 步骤 1: 研究信息
research_chain = LLMChain(
    llm=llm,
    prompt=ChatPromptTemplate.from_template(
        "搜索关于 {topic} 的最新信息"
    ),
    output_key="research"
)

# 步骤 2: 总结要点
summary_chain = LLMChain(
    llm=llm,
    prompt=ChatPromptTemplate.from_template(
        "总结以下信息,列出 {topic} 的 {n} 个要点:\n\n{research}"
    ),
    output_key="summary"
)

# 步骤 3: 生成文章
writing_chain = LLMChain(
    llm=llm,
    prompt=ChatPromptTemplate.from_template(
        "根据以下总结,写一篇关于 {topic} 的文章:\n\n{summary}"
    ),
    output_key="article"
)

# 创建工作流
workflow = SequentialChain(
    chains=[research_chain, summary_chain, writing_chain],
    input_variables=["topic", "n"],
    output_variables=["research", "summary", "article"]
)

# 运行
result = workflow(topic="人工智能", n=5)

print("研究:", result["research"])
print("\n总结:", result["summary"])
print("\n文章:", result["article"])
````

### 示例 2: 数据处理工作流

```python
from langchain.chains import SequentialChain
from langchain_community.vectorstores import FAISS
from langchain_openai import OpenAIEmbeddings

# 步骤 1: 加载数据
load_chain = TransformChain(
    output_variables=["data"],
    transform=lambda inputs: {"data": load_data(inputs["file_path"])}
)

# 步骤 2: 向量化
vectorize_chain = TransformChain(
    output_variables=["vectorstore"],
    transform=lambda inputs: {
        "vectorstore": FAISS.from_texts(
            texts=inputs["data"],
            embedding=OpenAIEmbeddings()
        )
    }
)

# 步骤 3: 检索信息
retrieve_chain = RetrievalQA.from_chain_type(
    llm=llm,
    retriever=lambda inputs: inputs["vectorstore"].as_retriever(),
    output_key="result"
)

# 创建工作流
workflow = SequentialChain(
    chains=[load_chain, vectorize_chain, retrieve_chain],
    input_variables=["file_path"],
    output_variables=["data", "result"]
)

# 运行
result = workflow(file_path="data.csv")

print("查询结果:", result["result"])
````

### 示例 3: 客服工作流

```python
from langchain.chains import RouterChain

# 定义不同类型的 Chain
greeting_chain = LLMChain(llm=llm, prompt="你好!有什么可以帮你?")

info_chain = LLMChain(
    llm=llm,
    prompt=ChatPromptTemplate.from_template(
        "回答问题: {question}\n\n知识库:\n{knowledge}"

    llm=llm,
    prompt=ChatPromptTemplate.from_template(
        "感谢您的反馈。我们会尽快处理。"

router = RouterChain(
    llm=llm,
    router_prompt="判断用户意图",
    chains={
        "greeting": greeting_chain,
        "info": info_chain,
        "complaint": complaint_chain
    }
)

# 运行
user_input = "我想了解产品价格"
result = router.run(user_input)

print(result)
````

## 🛠️ 高级工作流技巧

### 技巧 1: 条件执行

```python
from langchain.chains import Chain

class ConditionalChain(Chain):
    @property
    def input_keys(self):
        return ["condition", "data"]

    @property
    def output_keys(self):
        return ["result"]

    def _run(self, condition, data):
        if condition == "a":
            return process_a(data)
        elif condition == "b":
            return process_b(data)
        else:
            return process_c(data)
````

### 技巧 2: 并行执行

```python
from langchain.chains import RunnableParallel

# 并行执行多个 Chain
parallel_chain = RunnableParallel(
    chain1=chain1,
    chain2=chain2,
    chain3=chain3
)

# 同时运行
result = parallel_chain.run(input="data")
````

### 技巧 3: 循环执行

```python
from langchain.chains import Chain

class LoopChain(Chain):
    @property
    def input_keys(self):
        return ["input"]

    @property
    def output_keys(self):
        return ["output", "iterations"]

    def _run(self, input):
        output = input
        iterations = 0
        while not is_complete(output):
            output = process(output)
            iterations += 1
        return {"output": output, "iterations": iterations}
````

## 📊 工作流设计原则

### 1. 单一职责

```python
# ✅ 每个 Chain 只做一件事
search_chain = LLMChain(llm=llm, prompt="搜索信息")
sum_chain = LLMChain(llm=llm, prompt="总结信息")
write_chain = LLMChain(llm=llm, prompt="写文章")

# ❌ 一个 Chain 做太多事
general_chain = LLMChain(llm=llm, prompt="搜索、总结、写文章")
````

### 2. 清晰的输入输出

```python
# ✅ 明确的输入输出变量
workflow = SequentialChain(
    input_variables=["topic", "n"],
    output_variables=["research", "summary", "article"]
)

# ❌ 模糊的输入输出
workflow = SequentialChain(input_variables=["data"], output_variables=["result"])
````

### 3. 错误处理

```python
try:
    result = workflow.run(input="data")
except Exception as e:
    logger.error(f"工作流失败: {e}")
    # 回退方案
    result = fallback_chain.run(input="data")
````

## 💡 最佳实践

### 1. 合理的 Chain 数量

```python
# ✅ 3-5 个 Chain 组合
workflow = SequentialChain(chains=[chain1, chain2, chain3], ...)

# ❌ 过多的 Chain
workflow = SequentialChain(
    chains=[chain1, chain2, chain3, chain4, chain5, chain6],
    ...
)
````

### 2. 合理的参数传递

```python
# ✅ 只传递必要的参数
workflow = SequentialChain(
    input_variables=["topic"],
    ...
)

# ❌ 传递过多参数
workflow = SequentialChain(
    input_variables=["topic", "n", "language", "tone", "style", ...],
    ...
)
````

### 3. 适当的错误处理

```python
# 添加错误处理
try:
    result = workflow.run(input="data")
except Exception as e:
    # 记录错误
    logger.error(e)
    # 返回默认结果
    return default_result
````
