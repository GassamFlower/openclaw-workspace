## 💻 完整聊天机器人示例

### 基础聊天机器人

```python
from langchain_openai import ChatOpenAI
from langchain.chains import ConversationChain
from langchain.memory import ConversationBufferMemory

# 1. 创建 LLM
llm = ChatOpenAI(
    model="gpt-4",
    temperature=0.7
)

# 2. 创建记忆
memory = ConversationBufferMemory(
    input_key="input",
    output_key="output"
)

# 3. 创建对话链
conversation = ConversationChain(
    llm=llm,
    memory=memory,
    verbose=True
)

# 4. 聊天
print("聊天机器人已启动! (输入 'exit' 退出)")

while True:
    user_input = input("\n你: ")
    if user_input.lower() == 'exit':
        break

    response = conversation.predict(input=user_input)

    print("\n助手:", response)
````

## 🛠️ 不同记忆系统

### 1. Buffer Memory (完整历史)

```python
from langchain.memory import ConversationBufferMemory

memory = ConversationBufferMemory(
    return_messages=True  # 返回消息对象
)

# 记住所有对话
conversation = ConversationChain(
    llm=llm,
    memory=memory
)
````

### 2. Buffer Window Memory (最近 N 轮)

```python
from langchain.memory import ConversationBufferWindowMemory

# 只记住最近 4 轮对话
memory = ConversationBufferWindowMemory(
    k=4,  # 保留最近 4 轮
    return_messages=True
)

conversation = ConversationChain(
    llm=llm,
    memory=memory
)
````

### 3. Summary Memory (摘要)

```python
from langchain.memory import ConversationSummaryMemory

# 记住对话摘要
memory = ConversationSummaryMemory(
    llm=ChatOpenAI(model="gpt-4", temperature=0),
    return_messages=True
)

conversation = ConversationChain(
    llm=llm,
    memory=memory
)
````

## 🎮 带知识库的聊天机器人

### RAG + 记忆

```python
from langchain_community.vectorstores import FAISS
from langchain_openai import ChatOpenAI, OpenAIEmbeddings
from langchain.chains import ConversationalRetrievalChain
from langchain.memory import ConversationBufferMemory

# 1. 创建知识库
documents = [
    "LangChain 是一个 AI 应用开发框架",
    "LangChain 支持多种 LLM 提供商",
    "LangChain 提供工具调用和记忆功能",
]

vector_store = FAISS.from_texts(
    texts=documents,
    embedding=OpenAIEmbeddings(),
)

# 2. 创建记忆
memory = ConversationBufferMemory(
    memory_key="chat_history",
    return_messages=True
)

# 3. 创建 RAG Chain
llm = ChatOpenAI(model="gpt-4", temperature=0)

qa_chain = ConversationalRetrievalChain.from_llm(
    llm=llm,
    retriever=vector_store.as_retriever(),
    memory=memory,
    verbose=True
)

# 4. 聊天
print("聊天机器人已启动! (输入 'exit' 退出)")

while True:
    user_input = input("\n你: ")
    if user_input.lower() == 'exit':
        break

    result = qa_chain({"question": user_input})

    print("\n助手:", result["answer"])
````

## 📊 高级聊天机器人功能

from langchain_openai import ChatOpenAI
from langchain.chains import ConversationChain

### 功能 3: 知识库查询

```python
from langchain_community.vectorstores import Chroma
from langchain_openai import ChatOpenAI
from langchain.chains import ConversationalRetrievalChain
from langchain.memory import ConversationBufferMemory

# 创建知识库
documents = [
    "产品 A: 500元,性能好",
    "产品 B: 800元,功能多",
    "产品 C: 1000元,最强大",
]

vector_store = Chroma.from_documents(
    documents=documents,
    embedding=OpenAIEmbeddings(),
)

# 创建记忆
memory = ConversationBufferMemory()

# 创建 RAG Chain
llm = ChatOpenAI(model="gpt-4")

qa_chain = ConversationalRetrievalChain.from_llm(
    llm=llm,
    retriever=vector_store.as_retriever(),
    memory=memory
)

# 1. 询问产品
qa_chain.predict(input="有什么产品?")

# 2. 查询价格
qa_chain.predict(input="产品 A 多少钱?")

# 3. 比较产品
qa_chain.predict(input="哪个产品最便宜?")
````

## 💡 聊天机器人设计技巧

### 技巧 1: 设置系统提示词

```python
from langchain.prompts import ChatPromptTemplate, SystemMessagePromptTemplate, HumanMessagePromptTemplate

system_prompt = """你是一个友好的客服助手,专业回答用户问题。

规则:
1. 回答要简洁清晰
2. 遇到不知道的问题要诚实说明
3. 保持耐心和礼貌
4. 优先使用知识库中的信息
"""

human_prompt = "{input}"

messages = [
    SystemMessagePromptTemplate.from_template(system_prompt),
    HumanMessagePromptTemplate.from_template(human_prompt),
]

prompt = ChatPromptTemplate.from_messages(messages)

# 使用
conversation = ConversationChain(
    llm=llm,
    prompt=prompt,
    memory=memory
)
````

### 技巧 2: 限制回复长度

```python
from langchain_openai import ChatOpenAI

llm = ChatOpenAI(
    model="gpt-4",
    max_tokens=500  # 最多 500 个 token
)
````

### 技巧 3: 错误处理

```python
while True:
    try:
        user_input = input("你: ")
        if user_input.lower() == 'exit':
            break

        response = conversation.predict(input=user_input)
        print("\n助手:", response)

    except Exception as e:
        print(f"错误: {e}")
        print("请重试...")
````

## 🚀 Flask 部署

```python
from flask import Flask, request, jsonify
from chatbot import qa_chain

app = Flask(__name__)

@app.route('/chat', methods=['POST'])
def chat():
    data = request.json
    user_input = data.get('message', '')

    try:
        response = qa_chain.predict(input=user_input)
        return jsonify({"response": response})
    except Exception as e:
        return jsonify({"error": str(e)}), 500

if __name__ == '__main__':
    app.run(port=5000)
```
